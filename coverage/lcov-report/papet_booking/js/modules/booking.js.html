    var path = path || "";
    $.ajax( {
        "url":  path + "/data/data-test.json",
        "success" : function (data) {
            rooms = data.rooms;
            _.each(data.rooms, function (value, key) {
                $('select[data-rooms]').append($("<option>", {value : key, text: value.nom}))
            });
        },
        "error": function (data) {
            console.log(data);
        },
        "failure": function (data) {
            console.log(data);
        },
        "async": false
    });
    return rooms != null;
}
